/*Quick Sort*/
#include <stdio.h>
int partition(int a[],int p,int r)
{
    int temp,key,i,j;
    j=p;
    i=p-1;
    key=a[r];
    for(j=p;j<=r-1;j++)
    {
       if(a[j]<key)
       {
           i++;
           temp=a[i];
           a[i]=a[j];
           a[j]=temp;
       }
    }
    i=i+1;
    temp=a[i];
    a[i]=a[j];
    a[j]=temp;
    return i;
}
void quick_sort(int a[],int p,int r)
{
    int q;
    if (p<r)
    {
        q=partition(a,p,r);
        quick_sort(a,p,q-1);
        quick_sort(a,q+1,r);
    }
}
int main()
{
    int p=0,r,a[25],n,i;
    printf("Enter the number of elements\n");
    scanf("%d",&n);
    r=n-1;
    printf("\nEnter the elements\n");
    for(i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    quick_sort(a,p,r);
    printf("The sorted elements are:");
    for(i=0;i<n;i++)
    {
        printf("\t%d",a[i]);
    }
    return 0;
}
